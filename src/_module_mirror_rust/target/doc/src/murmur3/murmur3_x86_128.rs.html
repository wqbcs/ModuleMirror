9use std::io::{Read, Result};
10use std::ops::Shl;
11
12use crate::{copy_into_array, read_bytes};

14/// Use the x86 variant of the 128 bit murmur3 to hash some [Read] implementation.
15///
16/// # Example
17/// ```
18/// use std::io::Cursor;
19/// use murmur3::murmur3_x86_128;
20/// let hash_result = murmur3_x86_128(&mut Cursor::new("hello world"), 0);
21/// ```
22pub fn murmur3_x86_128<T: Read>(source: &mut T, seed: u32) -> Result<u128> {
23    const C1: u32 = 0x239b_961b;
24    const C2: u32 = 0xab0e_9789;
25    const C3: u32 = 0x38b3_4ae5;
26    const C4: u32 = 0xa1e3_8b93;
27    const C5: u32 = 0x561c_cd1b;
28    const C6: u32 = 0x0bca_a747;
29    const C7: u32 = 0x96cd_1c35;
30    const C8: u32 = 0x32ac_3b17;
31    const M: u32 = 5;
32
33    let mut h1: u32 = seed;
34    let mut h2: u32 = seed;
35    let mut h3: u32 = seed;
36    let mut h4: u32 = seed;
37
38    let mut buf = [0; 16];
39    let mut processed: usize = 0;
40    loop {
41        let read = read_bytes(source, &mut buf[..])?;
42        processed += read;
43        if read == 16 {
44            let k1 = u32::from_le_bytes(copy_into_array(&buf[0..4]));
45            let k2 = u32::from_le_bytes(copy_into_array(&buf[4..8]));
46            let k3 = u32::from_le_bytes(copy_into_array(&buf[8..12]));
47            let k4 = u32::from_le_bytes(copy_into_array(&buf[12..16]));
48            h1 ^= k1.wrapping_mul(C1).rotate_left(15).wrapping_mul(C2);
49            h1 = h1
50                .rotate_left(19)
51                .wrapping_add(h2)
52                .wrapping_mul(M)
53                .wrapping_add(C5);
54            h2 ^= k2.wrapping_mul(C2).rotate_left(16).wrapping_mul(C3);
55            h2 = h2
56                .rotate_left(17)
57                .wrapping_add(h3)
58                .wrapping_mul(M)
59                .wrapping_add(C6);
60            h3 ^= k3.wrapping_mul(C3).rotate_left(17).wrapping_mul(C4);
61            h3 = h3
62                .rotate_left(15)
63                .wrapping_add(h4)
64                .wrapping_mul(M)
65                .wrapping_add(C7);
66            h4 ^= k4.wrapping_mul(C4).rotate_left(18).wrapping_mul(C1);
67            h4 = h4
68                .rotate_left(13)
69                .wrapping_add(h1)
70                .wrapping_mul(M)
71                .wrapping_add(C8);
72        } else if read == 0 {
73            h1 ^= processed as u32;
74            h2 ^= processed as u32;
75            h3 ^= processed as u32;
76            h4 ^= processed as u32;
77            h1 = h1.wrapping_add(h2);
78            h1 = h1.wrapping_add(h3);
79            h1 = h1.wrapping_add(h4);
80            h2 = h2.wrapping_add(h1);
81            h3 = h3.wrapping_add(h1);
82            h4 = h4.wrapping_add(h1);
83            h1 = fmix32(h1);
84            h2 = fmix32(h2);
85            h3 = fmix32(h3);
86            h4 = fmix32(h4);
87            h1 = h1.wrapping_add(h2);
88            h1 = h1.wrapping_add(h3);
89            h1 = h1.wrapping_add(h4);
90            h2 = h2.wrapping_add(h1);
91            h3 = h3.wrapping_add(h1);
92            h4 = h4.wrapping_add(h1);
93            let x = ((h4 as u128) << 96) | ((h3 as u128) << 64) | ((h2 as u128) << 32) | h1 as u128;
94            return Ok(x);
95        } else {
96            let mut k1 = 0;
97            let mut k2 = 0;
98            let mut k3 = 0;
99            let mut k4 = 0;
100            if read >= 15 {
101                k4 ^= (buf[14] as u32).shl(16);
102            }
103            if read >= 14 {
104                k4 ^= (buf[13] as u32).shl(8);
105            }
106            if read >= 13 {
107                k4 ^= buf[12] as u32;
108                k4 = k4.wrapping_mul(C4).rotate_left(18).wrapping_mul(C1);
109                h4 ^= k4;
110            }
111            if read >= 12 {
112                k3 ^= (buf[11] as u32).shl(24);
113            }
114            if read >= 11 {
115                k3 ^= (buf[10] as u32).shl(16);
116            }
117            if read >= 10 {
118                k3 ^= (buf[9] as u32).shl(8);
119            }
120            if read >= 9 {
121                k3 ^= buf[8] as u32;
122                k3 = k3.wrapping_mul(C3).rotate_left(17).wrapping_mul(C4);
123                h3 ^= k3;
124            }
125            if read >= 8 {
126                k2 ^= (buf[7] as u32).shl(24);
127            }
128            if read >= 7 {
129                k2 ^= (buf[6] as u32).shl(16);
130            }
131            if read >= 6 {
132                k2 ^= (buf[5] as u32).shl(8);
133            }
134            if read >= 5 {
135                k2 ^= buf[4] as u32;
136                k2 = k2.wrapping_mul(C2).rotate_left(16).wrapping_mul(C3);
137                h2 ^= k2;
138            }
139            if read >= 4 {
140                k1 ^= (buf[3] as u32).shl(24);
141            }
142            if read >= 3 {
143                k1 ^= (buf[2] as u32).shl(16);
144            }
145            if read >= 2 {
146                k1 ^= (buf[1] as u32).shl(8);
147            }
148            if read >= 1 {
149                k1 ^= buf[0] as u32;
150            }
151            k1 = k1.wrapping_mul(C1);
152            k1 = k1.rotate_left(15);
153            k1 = k1.wrapping_mul(C2);
154            h1 ^= k1;
155        }
156    }
157}

159fn fmix32(k: u32) -> u32 {
160    const C1: u32 = 0x85eb_ca6b;
161    const C2: u32 = 0xc2b2_ae35;
162    const R1: u32 = 16;
163    const R2: u32 = 13;
164    let mut tmp = k;
165    tmp ^= tmp >> R1;
166    tmp = tmp.wrapping_mul(C1);
167    tmp ^= tmp >> R2;
168    tmp = tmp.wrapping_mul(C2);
169    tmp ^= tmp >> R1;
170    tmp
171}
