9use std::io::{Read, Result};
10
11use crate::read_bytes;
12
13const C1: u32 = 0x85eb_ca6b;
14const C2: u32 = 0xc2b2_ae35;
15const R1: u32 = 16;
16const R2: u32 = 13;
17const M: u32 = 5;
18const N: u32 = 0xe654_6b64;
19
20/// Use the 32 bit variant of murmur3 to hash some [Read] implementation.
21///
22/// # Example
23/// ```
24/// use std::io::Cursor;
25/// use murmur3::murmur3_32;
26/// let hash_result = murmur3_32(&mut Cursor::new("hello world"), 0);
27/// ```
28pub fn murmur3_32<T: Read>(source: &mut T, seed: u32) -> Result<u32> {
29    let mut buffer: [u8; 4] = [0; 4];
30    let mut processed = 0;
31    let mut state = seed;
32    loop {
33        match read_bytes(source, &mut buffer)? {
34            4 => {
35                processed += 4;
36                let k = u32::from_le_bytes(buffer);
37                state ^= calc_k(k);
38                state = state.rotate_left(R2);
39                state = (state.wrapping_mul(M)).wrapping_add(N);
40            }
41            3 => {
42                processed += 3;
43                let k: u32 =
44                    ((buffer[2] as u32) << 16) | ((buffer[1] as u32) << 8) | (buffer[0] as u32);
45                state ^= calc_k(k);
46            }
47            2 => {
48                processed += 2;
49                let k: u32 = ((buffer[1] as u32) << 8) | (buffer[0] as u32);
50                state ^= calc_k(k);
51            }
52            1 => {
53                processed += 1;
54                let k: u32 = buffer[0] as u32;
55                state ^= calc_k(k);
56            }
57            0 => return Ok(finish(state, processed)),
58            _ => panic!("Internal buffer state failure"),
59        }
60    }
61}
62
63fn finish(state: u32, processed: u32) -> u32 {
64    let mut hash = state;
65    hash ^= processed as u32;
66    hash ^= hash.wrapping_shr(R1);
67    hash = hash.wrapping_mul(C1);
68    hash ^= hash.wrapping_shr(R2);
69    hash = hash.wrapping_mul(C2);
70    hash ^= hash.wrapping_shr(R1);
71    hash
72}
73
74fn calc_k(k: u32) -> u32 {
75    const C1: u32 = 0xcc9e_2d51;
76    const C2: u32 = 0x1b87_3593;
77    const R1: u32 = 15;
78    k.wrapping_mul(C1).rotate_left(R1).wrapping_mul(C2)
79}
