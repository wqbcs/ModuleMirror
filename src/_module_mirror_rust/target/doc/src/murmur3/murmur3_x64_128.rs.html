9use std::io::{Read, Result};
10use std::ops::Shl;

12use crate::{copy_into_array, read_bytes};

14/// Use the x64 variant of the 128 bit murmur3 to hash some [Read] implementation.
15///
16/// # Example
17/// ```
18/// use std::io::Cursor;
19/// use murmur3::murmur3_x64_128;
20/// let hash_result = murmur3_x64_128(&mut Cursor::new("hello world"), 0);
21/// ```
22pub fn murmur3_x64_128<T: Read>(source: &mut T, seed: u32) -> Result<u128> {
23    const C1: u64 = 0x87c3_7b91_1142_53d5;
24    const C2: u64 = 0x4cf5_ad43_2745_937f;
25    const C3: u64 = 0x52dc_e729;
26    const C4: u64 = 0x3849_5ab5;
27    const R1: u32 = 27;
28    const R2: u32 = 31;
29    const R3: u32 = 33;
30    const M: u64 = 5;
31    let mut h1: u64 = seed as u64;
32    let mut h2: u64 = seed as u64;
33    let mut buf = [0; 16];
34    let mut processed: usize = 0;
35    loop {
36        let read = read_bytes(source, &mut buf[..])?;
37        processed += read;
38        if read == 16 {
39            let k1 = u64::from_le_bytes(copy_into_array(&buf[0..8]));
40            let k2 = u64::from_le_bytes(copy_into_array(&buf[8..]));
41            h1 ^= k1.wrapping_mul(C1).rotate_left(R2).wrapping_mul(C2);
42            h1 = h1
43                .rotate_left(R1)
44                .wrapping_add(h2)
45                .wrapping_mul(M)
46                .wrapping_add(C3);
47            h2 ^= k2.wrapping_mul(C2).rotate_left(R3).wrapping_mul(C1);
48            h2 = h2
49                .rotate_left(R2)
50                .wrapping_add(h1)
51                .wrapping_mul(M)
52                .wrapping_add(C4);
53        } else if read == 0 {
54            h1 ^= processed as u64;
55            h2 ^= processed as u64;
56            h1 = h1.wrapping_add(h2);
57            h2 = h2.wrapping_add(h1);
58            h1 = fmix64(h1);
59            h2 = fmix64(h2);
60            h1 = h1.wrapping_add(h2);
61            h2 = h2.wrapping_add(h1);
62            let x = ((h2 as u128) << 64) | (h1 as u128);
63            return Ok(x);
64        } else {
65            let mut k1 = 0;
66            let mut k2 = 0;
67            if read >= 15 {
68                k2 ^= (buf[14] as u64).shl(48);
69            }
70            if read >= 14 {
71                k2 ^= (buf[13] as u64).shl(40);
72            }
73            if read >= 13 {
74                k2 ^= (buf[12] as u64).shl(32);
75            }
76            if read >= 12 {
77                k2 ^= (buf[11] as u64).shl(24);
78            }
79            if read >= 11 {
80                k2 ^= (buf[10] as u64).shl(16);
81            }
82            if read >= 10 {
83                k2 ^= (buf[9] as u64).shl(8);
84            }
85            if read >= 9 {
86                k2 ^= buf[8] as u64;
87                k2 = k2.wrapping_mul(C2).rotate_left(33).wrapping_mul(C1);
88                h2 ^= k2;
89            }
90            if read >= 8 {
91                k1 ^= (buf[7] as u64).shl(56);
92            }
93            if read >= 7 {
94                k1 ^= (buf[6] as u64).shl(48);
95            }
96            if read >= 6 {
97                k1 ^= (buf[5] as u64).shl(40);
98            }
99            if read >= 5 {
100                k1 ^= (buf[4] as u64).shl(32);
101            }
102            if read >= 4 {
103                k1 ^= (buf[3] as u64).shl(24);
104            }
105            if read >= 3 {
106                k1 ^= (buf[2] as u64).shl(16);
107            }
108            if read >= 2 {
109                k1 ^= (buf[1] as u64).shl(8);
110            }
111            if read >= 1 {
112                k1 ^= buf[0] as u64;
113            }
114            k1 = k1.wrapping_mul(C1);
115            k1 = k1.rotate_left(31);
116            k1 = k1.wrapping_mul(C2);
117            h1 ^= k1;
118        }
119    }
120}

122fn fmix64(k: u64) -> u64 {
123    const C1: u64 = 0xff51_afd7_ed55_8ccd;
124    const C2: u64 = 0xc4ce_b9fe_1a85_ec53;
125    const R: u32 = 33;
126    let mut tmp = k;
127    tmp ^= tmp >> R;
128    tmp = tmp.wrapping_mul(C1);
129    tmp ^= tmp >> R;
130    tmp = tmp.wrapping_mul(C2);
131    tmp ^= tmp >> R;
132    tmp
133}
